REM-----------------------------------------------------------------
LET w=0
DO  WHILE  w<=8
   FOR  t=1  TO 8
      FOR i=1  TO 8
       
         IF  dist(i)>dist(t)+c(t,i)  THEN
            LET dist(i)=dist(t)+c(t,i)
            LET path$(i)=path$(t)&node$(i)
             
             
         END IF
          
      NEXT  i
       
   NEXT  t
   LET w=w+1
LOOP
